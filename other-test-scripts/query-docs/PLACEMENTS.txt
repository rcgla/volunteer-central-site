PLACEMENTS
----------
GET:
query($id: Int!) {
	placement (id: $id) {
		id
		confirmed
		dropped

		event {
			id
			name
			description
			start
			end
			visibility

		}
		role {
			id
			name

		}

		user {
			id
			name
			userType

		}

	}
}
